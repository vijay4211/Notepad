    case "50":

const filterByDiscount = (products, discountRange) => {
  return products.filter((product) => {
    const discount = calculateDiscount(product);
    return discountRange === "30"
      ? discount >= 30
      : discountRange === "50"
      ? discount >= 50
      : discountRange === "70"
      ? discount >= 70
      : true; // 'all' case
  });
};
